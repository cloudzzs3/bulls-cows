#I want a code where player 1 chooses a 3 digit number and player 2 has 10 tries to guess it if player 2 guesses the right
#number but its in the wrong place its a cow or a "c" and if they get the number in the right place and its right its a bull
#or a "b" and if player 2 guesses the number in under 10 tries it says "You have done it!" and if they can not guess it in
#at least 10 tries it will say "aww you lost player nice try!" and it will ask do you want to play again? player 2 can either
#say yes or no if they say yes the game will restart and a random 3 digit will appear on screen and the game will repeat until
#player 2 says "no" then it will say "Are you sure you want to quit?" if the player says "yes" then the game will quit and it will say
#"Goodbye thanks for playing" if the player says no the game will restart with a random number again.

def get_feedback(secret, guess):
    bulls = 0
    cows = 0

    for i in range(3):
        if guess[i] == secret[i]:
            bulls += 1
        elif guess[i] in secret:
            cows += 1

    return bulls, cows


def play_game():
    # Player 1 enters a 3-digit number
    secret_number = input("Player 1, enter a 3-digit number (unique digits): ")

    while len(secret_number) != 3 or not secret_number.isdigit() or len(set(secret_number)) != 3:
        print("Invalid input. Please enter a 3-digit number with unique digits.")
        secret_number = input("Player 1, enter a 3-digit number (unique digits): ")

    attempts = 10
    print("\nPlayer 2, you have 10 attempts to guess the number!")

    for attempt in range(attempts):
        guess = input(f"Attempt {attempt + 1}/{attempts}: Enter your guess (3 unique digits): ")

        if len(guess) != 3 or not guess.isdigit() or len(set(guess)) != 3:
            print("Invalid input. Please enter three unique digits.")
            continue

        bulls, cows = get_feedback(secret_number, guess)

        if bulls == 3:
            print("You have done it!")
            break
        else:
            print(f"{bulls} b, {cows} c")

    if bulls != 3:
        print("Aww, you lost player! Nice try! The correct number was:", secret_number)


def main():
    while True:
        play_game()

        play_again = input("\nDo you want to play again? (yes/no): ").strip().lower()
        if play_again != 'yes':
            confirm_quit = input("Are you sure you want to quit? (yes/no): ").strip().lower()
            if confirm_quit == 'yes':
                print("Goodbye! Thanks for playing.")
                break


if __name__ == "__main__":
    main()
